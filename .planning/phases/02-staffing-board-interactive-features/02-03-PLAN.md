    In `useDnd.ts`, the `handleDragEnd` currently handles only employee→project drops. Rewrite it to handle 3 scenarios:

    ```typescript
    const handleDragEnd = useCallback((event: DragEndEvent) => {
      const { active, over } = event

      const cleanup = () => {
        setActiveEmployeeId(null)
        setActiveAssignmentId(null, null)
      }

      if (!active.id || !over?.id) {
        cleanup()
        return
      }

      const dragType = active.data.current?.type as 'employee' | 'assignment' | undefined
      const dropType = over.data.current?.type as 'project' | 'home' | undefined

      // ─── SCENARIO 1: Employee → Project (EXISTING behavior) ───
      if ((dragType === 'employee' || !dragType) && dropType === 'project') {
        const activeEmployeeId = (active.data.current?.employeeId ?? active.id) as string
        const overProjectId = (over.data.current?.projectId ?? over.id) as string

        if (alreadyAssignedToProject(overProjectId, activeEmployeeId)) {
          notifications.show({
            title: "Assignment already exists",
            message: "Employee is already assigned to this project",
            color: "orange",
            autoClose: 2000
          })
          cleanup()
          return
        }

        createAssignment.mutateAsync({
          employeeId: activeEmployeeId,
          projectId: overProjectId,
          status: "hasNewProposedDispatch",
          id: uuidv7()
        })
        cleanup()
        return
      }

      // ─── SCENARIO 2: Assignment → Home (remove from project) ───
      if (dragType === 'assignment' && (dropType === 'home' || over.id === 'home')) {
        const assignmentId = active.data.current?.assignmentId as string
        const projectId = active.data.current?.projectId as string

        // Per CONTEXT.md: moveOutDate = today, no date picker prompt, no confirmation popup
        updateAssignment.mutate({
          where: { id: assignmentId, projectId },
          data: {
            status: "goingHome",
            moveOutDate: new Date()
          },
          metadata: {
            source: "dnd",
            component: "useDndAssignEmployee",
            description: "Employee dragged to Home - going home"
          }
        })
        cleanup()
        return
      }

      // ─── SCENARIO 3: Assignment → Different Project (copy) ───
      if (dragType === 'assignment' && dropType === 'project') {
        const sourceProjectId = active.data.current?.projectId as string
        const targetProjectId = (over.data.current?.projectId ?? over.id) as string
        const employeeId = active.data.current?.employeeId as string

        // Same-column drop: silent no-op per CONTEXT.md
        if (sourceProjectId === targetProjectId) {
          cleanup()
          return
        }

        // Duplicate check: employee already in target project
        if (alreadyAssignedToProject(targetProjectId, employeeId)) {
          notifications.show({
            title: "Assignment already exists",
            message: "Employee is already assigned to this project",
            color: "orange",
            autoClose: 2000
          })
          cleanup()
          return
        }

        // Per CONTEXT.md: true copy — employee stays in BOTH projects
        // All fields copied, only project_id changes. Status preserved.
        createAssignment.mutateAsync({
          employeeId: employeeId,
          projectId: targetProjectId,
          status: active.data.current?.status ?? "hasNewProposedDispatch",
          id: uuidv7()
        })
        cleanup()
        return
      }

      // Fallback: unrecognized drag/drop — cleanup
      cleanup()
    }, [setActiveEmployeeId, setActiveAssignmentId, createAssignment, updateAssignment, alreadyAssignedToProject])
    ```

    **Important notes:**
    - Scenario 1 preserves backward compatibility: if `dragType` is undefined (no type data — old code path), treat as employee drag.
    - Scenario 2 (Home): use `updateAssignment.mutate()` to set status="goingHome" and moveOutDate=today. Do NOT use `removeAssignment` — the card should remain visible with goingHome status (pink). The existing optimistic update in useStaffing handles this.
    - Scenario 3 (Copy): `createAssignment.mutateAsync()` creates a NEW assignment. Original stays. Per CONTEXT.md: "true copy (stays in both projects), all fields cloned, only project_id changes" and "status preserved". Pass the original status, not default.
    - `notifications` import should already exist in useDnd.ts.
    - Ensure `updateAssignment` is destructured from `useStaffing()` (it's already exported but may not be imported in useDnd).
    - No confirmation toast on success — per CONTEXT.md: "optimistic update IS the feedback"
    - No confirmation popup — per CONTEXT.md: no popups

    Also verify that `handleDragStart` and `handleDragCancel` (from Plan 01) correctly clear both active IDs.
