**1. Create `packages/api/src/funcs/dashboard/dashboard.funcs.ts`:**

Three functions:

a) `getDashboardRecentTickets(db, { userId, limit })`:
   - Find the user's profile by userId (to get profileId and aclRole)
   - Query `TksTicketsTable` for recent tickets relevant to the user:
     - For admin/manager roles (office-admin, project-manager, rpm, doo, divisional-manager, hr-manager, developer, office-admin-ticket-full-access, director-of-operations-with-inventory, office-admin-with-inventory, regional-project-manager-with-inventory): return the N most recent tickets across all departments, ordered by `createdAt` desc
     - For employee roles (employee, employee-with-projects, asst-project-manager-apm, ai-employee): return tickets where `createdBy` equals the user's profileId, ordered by `createdAt` desc
   - Include relations: `_creator` (columns: fullName, avatar), `_requestType` (columns: name), `_assignee` (columns: fullName)
   - Limit to `limit` (default 5)
   - Use Drizzle query API (db.query.TksTicketsTable.findMany)

b) `getDashboardMyAssignment(db, { userId })`:
   - Find profile by userId
   - Query `StaffingAssignmentsTable` for the user's ACTIVE assignments (where status is 'active' or 'confirmedIn' or 'confirmedOut' — check existing staffing status values)
   - Include the project relation to get project name, project number
   - Return the assignment(s) or null
   - Use db.query.StaffingAssignmentsTable.findMany with appropriate where clause
   - If user has no employeeInformation or is not an assignable employee, return empty array

c) `getDashboardStats(db, { userId })`:
   - Find profile by userId to determine role
   - For admin/manager roles: count of open tickets (status not 'closed'), count of active staffing projects
   - For employee roles: count of user's open tickets (createdBy = profileId, status not 'closed'), count of user's active assignments
   - Return `{ openTickets: number, activeProjects: number }`
   - Use Drizzle `count()` with appropriate filters

Follow the existing patterns from `tks_tickets.funcs.ts` and `staffing-refactor.funcs.ts`. Import types from `@repo/db/schema`. Use `DBClient` type from `@repo/db/client`.

import type { RoleConfig, WidgetConfig } from "../widgets/types"

// Widget definitions (reusable across configs)
const WIDGETS = {
  stats: {
    key: "stats",
    component: "StatsWidget",
    span: { base: 12, md: 12, lg: 12 },
  },
  recentTickets: {
    key: "recent-tickets",
    component: "RecentTicketsWidget",
    span: { base: 12, md: 6, lg: 6 },
  },
  recentActivity: {
    key: "recent-activity",
    component: "RecentActivityWidget",
    span: { base: 12, md: 6, lg: 6 },
  },
  myAssignment: {
    key: "my-assignment",
    component: "MyAssignmentWidget",
    span: { base: 12, md: 12, lg: 12 },
  },
  myTickets: {
    key: "my-tickets",
    component: "MyTicketsWidget",
    span: { base: 12, md: 12, lg: 12 },
  },
} as const satisfies Record<string, WidgetConfig>

// Role group configs
const MANAGEMENT_CONFIG: RoleConfig = {
  title: "Dashboard",
  subtitle: "Overview of tickets and staffing activity",
  widgets: [
    WIDGETS.stats,
    WIDGETS.recentTickets,
    WIDGETS.recentActivity,
  ],
}

const FIELD_LEAD_CONFIG: RoleConfig = {
  title: "Dashboard",
  subtitle: "Your projects and tickets",
  widgets: [
    WIDGETS.stats,
    WIDGETS.recentTickets,
    WIDGETS.myAssignment,
  ],
}

const EMPLOYEE_CONFIG: RoleConfig = {
  title: "Welcome Back",
  subtitle: "Your assignments and tickets",
  widgets: [
    WIDGETS.stats,
    WIDGETS.myAssignment,
    WIDGETS.myTickets,
  ],
}

// Map every role slug to a config
const MANAGEMENT_ROLES = [
  "office-admin",
  "project-manager",
  "rpm",
  "hr-manager",
  "doo",
  "divisional-manager",
  "developer",
  "office-admin-ticket-full-access",
  "director-of-operations-with-inventory",
  "office-admin-with-inventory",
  "regional-project-manager-with-inventory",
]

const FIELD_LEAD_ROLES = [
  "asst-project-manager-apm",
  "employee-with-projects",
]

// Build lookup
export const ROLE_CONFIGS: Record<string, RoleConfig> = {}

for (const role of MANAGEMENT_ROLES) {
  ROLE_CONFIGS[role] = MANAGEMENT_CONFIG
}
for (const role of FIELD_LEAD_ROLES) {
  ROLE_CONFIGS[role] = FIELD_LEAD_CONFIG
}
// employee and ai-employee use employee config
ROLE_CONFIGS["employee"] = EMPLOYEE_CONFIG
ROLE_CONFIGS["ai-employee"] = EMPLOYEE_CONFIG

export const DEFAULT_CONFIG = EMPLOYEE_CONFIG

/** Get role config with fallback */
export function getRoleConfig(roleSlug: string): RoleConfig {
  return ROLE_CONFIGS[roleSlug] ?? DEFAULT_CONFIG
}
