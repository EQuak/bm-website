**2. Create `apps/web/src/app/(application)/app/_components/WelcomeDashboard.tsx`:**

Client component that:
- Fetches profile via `api.profiles.getProfileByUserLogged.useSuspenseQuery()` (returns tuple `[profile]`)
- Calls `getRoleConfig(profile.aclRole)` to get widget configuration
- Renders personalized greeting: "Welcome back, {profile.firstName}!" with subtitle from config
- Renders widgets in a Mantine Grid using the config's widget list
- Resolves widget string keys to actual components using the WIDGET_COMPONENTS registry
- Sets up Supabase Realtime subscription for dashboard updates

```typescript
"use client"

import { useEffect } from "react"
import { Grid, Stack, Title, Text, Paper } from "@repo/mantine-ui"
import { api } from "#/trpc/react"
import { supabaseBrowserClient as supabase } from "#/utils/supabase/client"
import { TopPageWrapper } from "#/core/components/top-page-wrapper/TopPageWrapper"
import { getRoleConfig } from "./role-configs"
import { WIDGET_COMPONENTS } from "./widgets"

export function WelcomeDashboard() {
  const utils = api.useUtils()
  const [profile] = api.profiles.getProfileByUserLogged.useSuspenseQuery()
  const config = getRoleConfig(profile?.aclRole ?? "employee")

  // Real-time subscription for dashboard data freshness
  useEffect(() => {
    const setupRealtime = async () => {
      await supabase.realtime.setAuth()

      const channel = supabase
        .channel("dashboard_updates")
        .on(
          "postgres_changes",
          { event: "*", schema: "public", table: "tks_tickets" },
          () => {
            utils.dashboard.getRecentTickets.invalidate()
            utils.dashboard.getStats.invalidate()
          }
        )
        .on(
          "postgres_changes",
          { event: "*", schema: "public", table: "staffing_assignments" },
          () => {
            utils.dashboard.getMyAssignment.invalidate()
            utils.dashboard.getStats.invalidate()
            utils.staffing.getRecentStaffingActivity.invalidate()
          }
        )
        .subscribe((status, error) => {
          if (error) {
            console.error("Dashboard realtime error:", error)
          }
        })

      return () => {
        channel.unsubscribe()
      }
    }

    const cleanupPromise = setupRealtime()
    return () => {
      cleanupPromise.then((cleanup) => cleanup?.())
    }
  }, [utils])

  return (
    <TopPageWrapper pageTitle={config.title}>
      <Stack gap="lg">
        {/* Personalized greeting */}
        <div>
          <Title order={2}>
            Welcome back, {profile?.firstName ?? "User"}!
          </Title>
          <Text c="dimmed" size="sm">{config.subtitle}</Text>
        </div>

        {/* Widget grid */}
        <Grid gutter="md">
          {config.widgets.map(({ component: componentKey, span, key }) => {
            const WidgetComponent = WIDGET_COMPONENTS[componentKey]
            if (!WidgetComponent) return null
            return (
              <Grid.Col key={key} span={span}>
                <WidgetComponent />
              </Grid.Col>
            )
          })}
        </Grid>
      </Stack>
    </TopPageWrapper>
  )
}
```
